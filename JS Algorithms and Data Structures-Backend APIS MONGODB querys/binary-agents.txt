function binaryAgent(str) {
  let newStr = str.split(' '); 
  let result = "";
  for(let i = 0; i< newStr.length; i++){
    result +=String.fromCharCode(parseInt(newStr[i], 2))
  }
  console.log(result)
  return result;
}
